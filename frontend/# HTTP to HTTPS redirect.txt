# HTTP to HTTPS redirect
server {
    listen 80;
    server_name demoacademia.healthspire.org;

    if ($host = demoacademia.healthspire.org) {
        return 301 https://$host$request_uri;
    }
    return 404;
}

# Main server with HTTPS
server {
    listen 443 ssl;
    server_name demoacademia.healthspire.org;

    ssl_certificate /etc/letsencrypt/live/demoacademia.healthspire.org/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/demoacademia.healthspire.org/privkey.pem;
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;

    # Frontend static files serve
    root /var/www/sms/frontend/dist;
    index index.html;

    location / {
        try_files $uri $uri/ /index.html;
    }

    # API calls backend par proxy (localhost:5001)
    location /api/ {
        proxy_pass http://localhost:5001/;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
    }
}
